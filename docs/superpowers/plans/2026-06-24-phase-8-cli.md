```python
"""Test build_shortlist: rank theo fitness giam dan + decorrelate (loai candidate co
max|rho| voi cai da chon vuot nguong), pool-aware qua PoolCorrelation."""

from __future__ import annotations

import numpy as np

from src.backtest.metrics_local import AlphaMetrics
from src.backtest.pool_corr import PoolCorrelation
from src.pipeline.shortlist import ShortlistCandidate, build_shortlist


def _dates(start: str, n: int) -> np.ndarray:
    return (np.datetime64(start) + np.arange(n)).astype("datetime64[D]")


def _metrics(fitness: float) -> AlphaMetrics:
    return AlphaMetrics(
        sharpe=1.0, annual_return=0.1, turnover=0.3, max_drawdown=0.05,
        fitness=fitness, per_year_sharpe={2021: 1.0}, weight_concentration=0.05,
    )


def test_ranks_by_fitness_descending_when_uncorrelated():
    dates = _dates("2021-01-01", 20)
    rng = np.random.default_rng(0)
    low = ShortlistCandidate("low", _metrics(0.5), rng.normal(size=20), dates)
    high = ShortlistCandidate("high", _metrics(2.0), rng.normal(size=20), dates)
    out = build_shortlist([low, high], top_k=2, max_corr=0.7)
    assert [c.expr for c in out] == ["high", "low"]


def test_decorrelate_drops_high_correlation_pair():
    dates = _dates("2021-01-01", 20)
    base_pnl = np.linspace(0.01, 0.20, 20)
    a = ShortlistCandidate("a_best", _metrics(2.0), base_pnl.copy(), dates)
    # b co pnl ty le voi a -> rho = 1.0, phai bi loai du fitness thap hon a nhung neu
    # b duoc xet truoc trong thu tu khong-lien-quan thi van phai loai vi trung voi a.
    b = ShortlistCandidate("b_dup", _metrics(1.5), base_pnl.copy() * 2.0, dates)
    c = ShortlistCandidate("c_diff", _metrics(1.0), -base_pnl.copy(), dates)
    out = build_shortlist([a, b, c], top_k=3, max_corr=0.7)
    names = [cand.expr for cand in out]
    assert "a_best" in names
    assert "b_dup" not in names  # trung voi a_best (rho=1.0 >= 0.7) -> bi loai
    assert "c_diff" in names  # rho=-1.0 voi a -> CUNG bi loai vi |rho| >= 0.7
    assert len(out) == 1


def test_respects_top_k_limit():
    dates = _dates("2021-01-01", 20)
    rng = np.random.default_rng(1)
    cands = [
        ShortlistCandidate(f"x{i}", _metrics(float(i)), rng.normal(size=20), dates)
        for i in range(5)

    out = build_shortlist(cands, top_k=2, max_corr=0.99)
    assert len(out) == 2
    assert out[0].expr == "x4"  # fitness cao nhat
    assert out[1].expr == "x3"

    pool_corr = PoolCorrelation(pool={1: (dates, pool_pnl.copy())})
    matches_pool = ShortlistCandidate("dup_pool", _metrics(2.0), pool_pnl.copy() * 3.0, dates)
    fresh = ShortlistCandidate("fresh", _metrics(1.0), -pool_pnl.copy(), dates)
    out = build_shortlist([matches_pool, fresh], top_k=2, max_corr=0.7, pool_corr=pool_corr)
    names = [c.expr for c in out]
    assert "dup_pool" not in names
    assert "fresh" not in names  # |rho|=1.0 voi pool -> cung bi loai

def test_does_not_mutate_input_list():

        ShortlistCandidate("a", _metrics(1.0), np.ones(10), dates),
        ShortlistCandidate("b", _metrics(2.0), np.ones(10) * -1, dates),

    original_order = [c.expr for c in cands]
    build_shortlist(cands, top_k=2, max_corr=0.99)
```

```python
# src/pipeline/shortlist.py
"""Rank + decorrelate candidate da co metrics/pnl -> short-list cuoi cung de sim Brain.

Day la buoc tong hop cua B14 (end-to-end pipeline): nhan candidate da qua score_one/
generate_many, xep hang theo fitness, roi loai tuan tu cac candidate tuong quan PnL qua
cao voi cai DA CHON (decorrelate noi bo) VA voi pool da pass (decorrelate pool-aware, qua
PoolCorrelation) - dung dung nguyen tac B9: PnL self-corr la nguyen nhan reject hang dau,
khong phai AST-hash dedup.
"""

from __future__ import annotations

from dataclasses import dataclass

import numpy as np
import numpy.typing as npt

from src.backtest.metrics_local import AlphaMetrics
from src.backtest.pool_corr import PoolCorrelation
from src.local_types import Dates


@dataclass(frozen=True, slots=True)
class ShortlistCandidate:
    """Mot candidate da backtest xong: expr + metrics + PnL hang ngay (de tinh tuong quan)."""

    expr: str
    metrics: AlphaMetrics
    pnl: npt.NDArray[np.float64]
    dates: Dates


def _pairwise_abs_rho(
    pnl_a: npt.NDArray[np.float64],
    dates_a: Dates,
    pnl_b: npt.NDArray[np.float64],
    dates_b: Dates,
) -> float | None:
    """Pearson |rho| tren giao ngay chung; None neu khong du diem/phuong sai (giong logic
    PoolCorrelation._pairwise_rho, Phase 6) - khong bia rho=0 gia."""
    common = np.intersect1d(dates_a, dates_b)
    if common.size < 2:
        return None
    idx_a = np.searchsorted(dates_a, common)
    idx_b = np.searchsorted(dates_b, common)
    x = pnl_a[idx_a]
    y = pnl_b[idx_b]
    finite = np.isfinite(x) & np.isfinite(y)
    if finite.sum() < 2:
        return None
    x = x[finite]
    y = y[finite]
    if np.std(x) == 0.0 or np.std(y) == 0.0:
        return None
    rho = float(np.corrcoef(x, y)[0, 1])
    if np.isnan(rho):
        return None
    return abs(rho)


def build_shortlist(
    candidates: list[ShortlistCandidate],
    top_k: int,
    max_corr: float,
    pool_corr: PoolCorrelation | None = None,
) -> list[ShortlistCandidate]:
    """Xep hang `candidates` theo `metrics.fitness` giam dan, roi quet tuan tu: giu candidate
    neu max|rho| voi MOI candidate da giu truoc do VA voi pool (qua `pool_corr.max_corr` neu
    co) deu < `max_corr`. Dung khi du `top_k` hoac het candidates. Khong sua doi danh sach
    dau vao (tra ve list moi)."""
    ranked = sorted(candidates, key=lambda c: c.metrics.fitness, reverse=True)
    kept: list[ShortlistCandidate] = []
    for cand in ranked:
        if len(kept) >= top_k:
            break
        if pool_corr is not None:
            pool_rho, _worst_id = pool_corr.max_corr(cand.pnl, cand.dates)
            if pool_rho >= max_corr:
                continue
        too_correlated = False
        for chosen in kept:
            rho = _pairwise_abs_rho(cand.pnl, cand.dates, chosen.pnl, chosen.dates)
            if rho is not None and rho >= max_corr:
                too_correlated = True
                break
        if too_correlated:
            continue
        kept.append(cand)
    return kept
```

```python
"""Lop orchestration cuoi cung cua MiniBrain (B14): score_one cham 1 expr KHONG dot sim;
generate_many drive GPEngine roi rut short-list. Network-agnostic - nhan MarketData/
repository/GPEngine qua tham so injected, test duoc bang fake hoan toan."""

from __future__ import annotations

from dataclasses import dataclass

import numpy as np
import numpy.typing as npt

from src.backtest.backtester import Backtester
from src.backtest.config import PortfolioConfig
from src.backtest.gates import GateEvaluator, GateVerdict
from src.backtest.metrics_local import AlphaMetrics, MetricsCalculator
from src.backtest.pool_corr import PoolCorrelation
from src.backtest.portfolio import PortfolioBuilder
from src.data.market_panel import MarketData
from src.engine.evaluator import EvalContext, Evaluator
from src.lang.parser import ParseError, parse
from src.lang.registry import default_registry
from src.lang.visitors import DepthVisitor, FieldCollector
from src.local_types import Dates

_EMPTY_METRICS = AlphaMetrics(
    sharpe=0.0, annual_return=0.0, turnover=0.0, max_drawdown=0.0,
    fitness=0.0, per_year_sharpe={}, weight_concentration=0.0,
)


def score_one(
    expr: str,
    cfg: PortfolioConfig,
    data: MarketData,
    pool: dict[int, tuple[Dates, npt.NDArray[np.float64]]] | None = None,
) -> tuple[AlphaMetrics, GateVerdict]:
    """parse -> eval -> portfolio -> backtest -> metrics -> pool_corr -> gate.

    Thuan local, khong goi mang/sim Brain. Bat dinh: cung (expr, cfg, data, pool) -> cung
    ket qua. Loi parse/eval -> AlphaMetrics rong + GateVerdict that bai voi ly do ro rang
    (khong silent, khong bia metrics)."""
    try:
        node = parse(expr)
    except ParseError as exc:
        return _EMPTY_METRICS, GateVerdict(
            passed=False, hard_failures=[f"parse lỗi: {exc}"],
        )

    fields = FieldCollector().visit(node)
    fields_ok = all(name in data.fields or name in ("close", "returns") for name in fields)
    depth = DepthVisitor().visit(node)

    registry = default_registry()
    ctx = EvalContext(data=data, registry=registry, cache=None)
    try:
        signal = Evaluator(ctx).evaluate(node)
    except (KeyError, ValueError) as exc:
        return _EMPTY_METRICS, GateVerdict(
            passed=False, hard_failures=[f"eval lỗi: {exc}"],
        )

    if np.all(np.isnan(signal)):
        return _EMPTY_METRICS, GateVerdict(
            passed=False, hard_failures=["signal toàn NaN — không có giá trị dùng được"],
        )

    weights = PortfolioBuilder().build(signal, cfg, data)
    bt = Backtester().run(weights, data)
    metrics = MetricsCalculator().compute(bt, data)

    if pool:
        pool_corr = PoolCorrelation(pool=pool)
        verdict = GateEvaluator().evaluate_with_pool(
            metrics,
            candidate_pnl=bt.daily_pnl,
            candidate_dates=data.dates,
            pool_corr=pool_corr,
            depth=depth,
            fields_ok=fields_ok,
        )
    else:
        verdict = GateEvaluator().evaluate(
            metrics, self_corr=0.0, depth=depth, fields_ok=fields_ok,
        )

    return metrics, verdict
```

```python
from src.lang.visitors import Serializer  # them vao khoi import dau file
from src.pipeline.shortlist import ShortlistCandidate, build_shortlist  # them vao khoi import


def generate_many(
    gp_engine: object,
    generations: int,
    cfg: PortfolioConfig,
    data: MarketData,
    top_k: int,
    max_corr: float,
    pool: dict[int, tuple[Dates, npt.NDArray[np.float64]]] | None = None,
) -> list[ShortlistCandidate]:

    nguon AlphaMetrics duy nhat cho ca B14), roi build_shortlist top_k + decorrelate
    pool-aware. Individual.fitness is None (chua eval trong GP) bi bo qua."""

    serializer = Serializer()

    candidates: list[ShortlistCandidate] = []
    pool_corr = PoolCorrelation(pool=pool) if pool else None
    for ind in individuals:
        if ind.fitness is None:

        candidates.append(
            ShortlistCandidate(expr=expr_str, metrics=metrics, pnl=bt.daily_pnl, dates=data.dates)
        )
```

```python
def test_score_one_with_real_panel_prints_metrics_and_verdict(tmp_path, small_panel):

    assert "passed" in result.stdout.lower() or "verdict" in result.stdout.lower()
```

```python
def _load_market_data_or_exit(market_data_dir: str):
    """Doc MarketData tu thu muc parquet; bao loi ro rang (khong silent) neu thieu/rong."""
    from pathlib import Path

    data_path = Path(market_data_dir)
    if not data_path.exists() or not any(data_path.iterdir()):
        console.print(
            f"[red]Không có market data tại '{market_data_dir}' — chạy "
            "market_fetch (Phase 0) trước, hoặc trỏ --market-data-dir đúng thư mục "
            "parquet.[/red]"
        )
        raise typer.Exit(code=1)
    from src.data.adapters.parquet_source import ParquetSource

    return ParquetSource(data_path).load(start="", end="", universe="TOP3000")


@app.command("score-one")
def score_one_cmd(
    expr: str = typer.Argument(..., help="Biểu thức FASTEXPR cần chấm điểm local"),
    market_data_dir: str = typer.Option(..., "--market-data-dir", help="Thư mục parquet MarketData (Phase 0)"),
    delay: int = typer.Option(1, help="Delay áp dụng trong PortfolioConfig"),
) -> None:
    """Chấm 1 biểu thức local (parse->eval->backtest->metrics->gate) — không đốt sim Brain."""
    _setup_logging()
    data = _load_market_data_or_exit(market_data_dir)
    cfg = PortfolioConfig(delay=delay)
    metrics, verdict = score_one(expr, cfg, data)

    table = Table(title=f"score-one: {expr}")
    table.add_column("Metric")
    table.add_column("Giá trị", justify="right")
    table.add_row("sharpe", f"{metrics.sharpe:.4f}")
    table.add_row("annual_return", f"{metrics.annual_return:.4f}")
    table.add_row("turnover", f"{metrics.turnover:.4f}")
    table.add_row("max_drawdown", f"{metrics.max_drawdown:.4f}")
    table.add_row("fitness", f"{metrics.fitness:.4f}")
    table.add_row("weight_concentration", f"{metrics.weight_concentration:.4f}")
    console.print(table)
    console.print(f"verdict.passed = [bold]{verdict.passed}[/bold]")
    if verdict.hard_failures:
        console.print(f"[red]hard_failures: {verdict.hard_failures}[/red]")
    if verdict.soft_scores:
        console.print(f"soft_scores: {verdict.soft_scores}")


@app.command("generate")
def generate_cmd(
    market_data_dir: str = typer.Option(..., "--market-data-dir", help="Thư mục parquet MarketData (Phase 0)"),
    generations: int = typer.Option(10, help="Số thế hệ GP"),
    top_k: int = typer.Option(20, help="Số candidate giữ lại trong short-list"),
    max_corr: float = typer.Option(0.70, help="Ngưỡng |rho| decorrelate (mặc định = SELF_CORR_MAX)"),
    delay: int = typer.Option(1, help="Delay áp dụng trong PortfolioConfig"),
    out_file: str = typer.Option("", "--out", help="Ghi short-list ra file CSV; rỗng = chỉ in console"),
) -> None:
    """Chạy GPEngine sinh candidate, xếp hạng + decorrelate, xuất short-list để sim Brain."""
    _setup_logging()
    data = _load_market_data_or_exit(market_data_dir)
    cfg = PortfolioConfig(delay=delay)

    from src.gp.engine import GPEngine  # import tai cho - tranh phu thuoc cung luc khoi dong CLI

    gp_engine = GPEngine.from_market_data(data)  # XAC NHAN constructor thuc te Phase 7 truoc khi dung
    shortlist = generate_many(
        gp_engine=gp_engine, generations=generations, cfg=cfg, data=data,
        top_k=top_k, max_corr=max_corr,
    )

    table = Table(title=f"Short-list ({len(shortlist)} candidate)")
    table.add_column("Expr")
    table.add_column("Sharpe", justify="right")
    table.add_column("Fitness", justify="right")
    for cand in shortlist:
        table.add_row(cand.expr, f"{cand.metrics.sharpe:.4f}", f"{cand.metrics.fitness:.4f}")
    console.print(table)

    if out_file:
        import csv

        with open(out_file, "w", newline="", encoding="utf-8") as f:
            writer = csv.writer(f)
            writer.writerow(["expr", "sharpe", "fitness", "turnover"])
            for cand in shortlist:
                writer.writerow(
                    [cand.expr, cand.metrics.sharpe, cand.metrics.fitness, cand.metrics.turnover]
                )
        console.print(f"[green]Đã ghi short-list ra {out_file}[/green]")
```
